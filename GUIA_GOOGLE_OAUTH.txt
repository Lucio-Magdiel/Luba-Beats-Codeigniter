📍 PASO 1: CREAR PROYECTO
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Ir a: https://console.cloud.google.com/
2. Iniciar sesión con tu cuenta de Google
3. Click en el selector de proyecto (arriba izquierda)
4. Click en "NUEVO PROYECTO"
5. Nombre: LubaBeats-Beta
6. Click "CREAR"
7. Esperar a que se cree el proyecto

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📍 PASO 2: HABILITAR API
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Menú lateral → "APIs y servicios" → "Biblioteca"
2. Buscar: "Google+ API"
3. Click en "Google+ API"
4. Click en "HABILITAR"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📍 PASO 3: PANTALLA DE CONSENTIMIENTO
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Menú lateral → "APIs y servicios" → "Pantalla de consentimiento"
2. Seleccionar: "Externo"
3. Click "CREAR"

   PÁGINA 1 - INFORMACIÓN DE LA APP:
   ┌─────────────────────────────────────────────────┐
   │ Nombre de la aplicación: LubaBeats Beta         │
   │ Correo de asistencia: tu_correo@gmail.com       │
   │ Logo: (opcional)                                 │
   │ Dominios autorizados: (dejar vacío)             │
   │ Correo del desarrollador: tu_correo@gmail.com   │
   └─────────────────────────────────────────────────┘
   Click "GUARDAR Y CONTINUAR"

   PÁGINA 2 - ALCANCES:
   ┌─────────────────────────────────────────────────┐
   │ Click "AGREGAR O QUITAR ALCANCES"               │
   │ Seleccionar:                                     │
   │   ✅ .../auth/userinfo.email                    │
   │   ✅ .../auth/userinfo.profile                  │
   │   ✅ openid                                      │
   └─────────────────────────────────────────────────┘
   Click "ACTUALIZAR" → "GUARDAR Y CONTINUAR"

   PÁGINA 3 - USUARIOS DE PRUEBA:
   ┌─────────────────────────────────────────────────┐
   │ Click "AGREGAR USUARIOS"                        │
   │ Ingresar: tu_correo@gmail.com                   │
   │ Click "AGREGAR"                                  │
   └─────────────────────────────────────────────────┘
   Click "GUARDAR Y CONTINUAR"

   PÁGINA 4 - RESUMEN:
   Click "VOLVER AL PANEL"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

📍 PASO 4: CREAR CREDENCIALES (¡EL MÁS IMPORTANTE!)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Menú lateral → "APIs y servicios" → "Credenciales"
2. Click "CREAR CREDENCIALES" → "ID de cliente de OAuth"
3. Tipo: "Aplicación web"

   │ │ http://localhost/CHOJIN/public/auth/google/callback  │ │

   │ │ callback                                              │ │
   │ └──────────────────────────────────────────────────────┘ │
   └──────────────────────────────────────────────────────────┘

📍 PASO 6: CONFIGURAR EN TU PROYECTO
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Abrir archivo: c:\xampp\htdocs\CHOJIN\env

2. Buscar estas líneas:

   google.clientId = TU_CLIENT_ID_AQUI
   google.clientSecret = TU_CLIENT_SECRET_AQUI

3. Reemplazar con tus credenciales:

   google.clientId = 123456789-abc123xyz.apps.googleusercontent.com
   google.clientSecret = GOCSPX-abcd1234efgh5678ijkl

4. GUARDAR el archivo

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

3. Seleccionar tu cuenta de Google

4. Autorizar la aplicación

5. ¡Deberías estar dentro! ✅

🐛 ERRORES COMUNES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

❌ Error: redirect_uri_mismatch
   → La URI en Google Console no coincide exactamente
   → Verifica que NO falte el /callback al final
   → Verifica http vs https

❌ Error: Access blocked
   → No configuraste la pantalla de consentimiento
   → Tu correo no está en usuarios de prueba

❌ Error: invalid_client
   → Client ID o Secret incorrectos
   → Verifica que no haya espacios al copiar
